translate("Install Bibledit Cloud on Ubuntu")

translate("Follow the steps below to install Bibledit.") translate("It needs basic knowledge about working with Linux servers.")

  1. translate("Open a terminal.")

  2. translate("Update the software sources:")

    sudo apt-get update

  3. translate("Install the software Bibledit relies on:")

    sudo apt-get install build-essential git zip pkgconf libcurl4-openssl-dev libssl-dev poppler-utils libsword-utils diatheke default-jre-headless ldap-utils

  4. translate("Download Bibledit:")

    wget http://bibledit.org/cloud/bibledit-##cloudversion##.tar.gz

  5. translate("Unpack Bibledit in folder bibledit:")

    mkdir -p bibledit

    tar xf bibledit-##cloudversion##.tar.gz -C bibledit --strip-components=1

  6. translate("Change to the directory:")

    cd bibledit

  7. translate("Build Bibledit:")

    ./configure

    make

    translate("This takes a while.") translate("There will be many technical messages, but there should be no errors.")

  8. translate("Start Bibledit:")

    ./bibledit &

  9. translate("Bibledit will be accessible through the IP address port 8080.") translate("Examples:")

    http://website.org:8080

    http://192.168.2.1:8080

    http://localhost:8080

  10. translate("Open the web address in the browser, follow the instructions on the screen to configure Bibledit, and log in.")
